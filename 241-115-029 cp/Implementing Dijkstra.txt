#include <bits/stdc++.h>
using namespace std;

int main() {
    int V;
    cin >> V;

    
    vector<vector<int>> graph(V, vector<int>(V));

    
    
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) {
            cin >> graph[i][j];
        }
    }

    int src;
    cin >> src;

    vector<int> dist(V, INT_MAX);
    vector<bool> visited(V, false);

    dist[src] = 0;

    for (int count = 0; count < V - 1; count++) {

     
        int u = -1;
        int mn = INT_MAX;

        for (int i = 0; i < V; i++) {
            if (!visited[i] && dist[i] < mn) {
                mn = dist[i];
                u = i;
            }
        }

        visited[u] = true;

        for (int v = 0; v < V; v++) {

            if (!visited[v] && graph[u][v] != 0 &&
                dist[u] != INT_MAX &&
                dist[u] + graph[u][v] < dist[v]) 
            {
                dist[v] = dist[u] + graph[u][v];
            }
        }
    }

    
    for (int i = 0; i < V; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;

    return 0;
}
