#include <iostream>
#include <vector>
using namespace std;

int binarySearch(const vector<int>& arr, int x) {
    int l = 0, r = arr.size() - 1;

    while (l <= r) {
        int m = l + (r - ) / 2;

        if (arr[m] == x) {
            return m; 
        } else if (arr[m] < x) {
            l = m + 1; 
        } else {
            r = m - 1; 
        }
    }

    return -1; 
}

int main() {
    int n;
    cin >> n;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    int X;
    cin >> X;

    int index = binarySearch(arr, X);
    cout << index << endl;

    return 0;
}
