#include <iostream>
#include <vector>
using namespace std;

int lowerBound(const vector<int>& nums, int x) {
    int l = 0, r = nums.size(); 

    while (l < r) {
        int mid = l + (r - l) / 2;

        if (nums[mid] < x) {
            l = mid + 1; 
        } else {
            r = mid; 
        }
    }

    return l; 
}

int main() {
    int n;
    cin >> n;
    vector<int> nums(n);
    for (int i = 0; i < n; i++) {
        cin >> nums[i];
    }
    int x;
    cin >> x;

    cout << lowerBound(nums, x) << endl;

    return 0;
}
